    ```python
    import logging
    from collections import deque
    from typing import Optional
    import numpy as np

    logger = logging.getLogger("sim.mean_field.bifurcation")

    class BifurcationDetector:
        """Detect bifurcation events via Jacobian eigenvalue lambda_1 spikes (per D-01)."""

        def __init__(
            self,
            agent_name: str,
            lambda_threshold: float = -0.1,
            spike_min_separation: int = 10,
        ):
            self.agent_name = agent_name
            self.lambda_threshold = lambda_threshold
            self.spike_min_separation = spike_min_separation
            self._buffer: deque = deque(maxlen=3)  # stores last 3 Re(lambda_1) values
            self._last_fire_tick: Optional[int] = None
            self.events: list[dict] = []

        def compute_jacobian(self, mf: "MeanFieldSystem") -> np.ndarray:
            """Compute z-subspace Jacobian per D-02.

            J(z) = -I + (u - s) * diag(sech^2((u - s)*M@z + b - beta)) * M

            Standard model: s = 0 (mf.g_adapt == 0.0)
            SFA model: s = mf.adapt_ring (element-wise)
            """
            z = mf.neural_ring
            M = mf.M
            b = mf.b
            u = mf.u
            beta = mf.beta
            n = mf.num_neurons

            # s is scalar 0 for standard, or the adapt_ring vector for SFA
            s = mf.adapt_ring if mf.g_adapt > 0.0 else 0.0

            # Argument to sech^2: (u - s) * M @ z + b - beta
            arg = (u - s) * (M @ z) + b - beta
            # sech^2(x) = 1 / cosh^2(x)
            sech2 = 1.0 / np.cosh(arg) ** 2

            # J = -I + (u - s) * diag(sech2) @ M
            J = -np.eye(n) + (u - s) * np.diag(sech2) @ M
            # When s is a vector, (u - s) is element-wise and multiplies each row
            # But diag(sech2) is already element-wise, so:
            # J_ij = -delta_ij + (u - s_i) * sech2_i * M_ij
            # This is correctly computed as:
            # J = -I + diag((u - s) * sech2) @ M
            if isinstance(s, np.ndarray):
                J = -np.eye(n) + np.diag((u - s) * sech2) @ M
            else:
                J = -np.eye(n) + (u - s) * np.diag(sech2) @ M

            return J

        def compute_lambda1(self, mf: "MeanFieldSystem") -> float:
            """Return Re(lambda_1) - largest real part of Jacobian eigenvalues."""
            J = self.compute_jacobian(mf)
            eigenvalues = np.linalg.eigvals(J)
            return float(np.max(eigenvalues.real))

        def update(
            self,
            tick: int,
            mf: "MeanFieldSystem",
            bump_angle: float,
            target_angles: list[float],
            target_ids: list[str],
        ) -> Optional[dict]:
            """Check for bifurcation at this tick. Returns event dict if detected, else None.

            Per D-03: uses 3-sample micro-buffer for local-maximum spike detection.
            Per D-04: suppresses re-detection for spike_min_separation ticks.
            Per D-05: event dict has keys {agent, tick, lambda1, target}.
            """
            lambda1 = self.compute_lambda1(mf)
            self._buffer.append((tick, lambda1))

            if len(self._buffer) < 3:
                return None

            (t_prev, v_prev), (t_curr, v_curr), (t_next, v_next) = self._buffer
            # Local maximum check + threshold check (D-03)
            if v_prev < v_curr > v_next and v_curr > self.lambda_threshold:
                # Suppression check (D-04)
                if (self._last_fire_tick is not None
                        and t_curr - self._last_fire_tick < self.spike_min_separation):
                    return None

                # Find nearest target (D-05)
                nearest_target = self._nearest_target(bump_angle, target_angles, target_ids)

                event = {
                    "agent": self.agent_name,
                    "tick": t_curr,
                    "lambda1": round(v_curr, 6),
                    "target": nearest_target,
                }
                self.events.append(event)
                self._last_fire_tick = t_curr
                logger.info(
                    "Bifurcation detected: agent=%s tick=%d lambda1=%.6f target=%s",
                    self.agent_name, t_curr, v_curr, nearest_target,
                )
                return event

            return None

        @staticmethod
        def _nearest_target(
            bump_angle: float,
            target_angles: list[float],
            target_ids: list[str],
        ) -> str:
            """Return the target ID closest to the current bump angle."""
            if not target_angles or not target_ids:
                return "unknown"
            deltas = np.abs(
                (np.array(target_angles) - bump_angle + np.pi) % (2 * np.pi) - np.pi
            )
            idx = int(np.argmin(deltas))
            return target_ids[idx] if idx < len(target_ids) else "unknown"
    ```

    **Key design points:**
    - `compute_jacobian()` uses the exact formula from D-02. For SFA (g_adapt > 0), `s = mf.adapt_ring` (a vector); for standard model, `s = 0.0` (scalar). Both paths produce the correct Jacobian via `diag((u - s) * sech2) @ M`.
    - `_buffer` stores `(tick, lambda1)` tuples so the spike detection in `update()` can reference the tick of the peak (D-03 says `bifurcation_detected(tick=current_tick - 1)` because `curr` is from the previous call).
    - `_nearest_target()` uses circular distance (same as `_delta_angle` pattern in mean_field_systems.py).
    - Logger: `sim.mean_field.bifurcation` per conventions.
